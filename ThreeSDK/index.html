ThreeSDK ​

基于 three 0.151.1 封装的工具类

node 环境> node@18.20.0

安装：npm 地址

sh
cnpm i -S three-plugins-fn
1
初始化插件 ​
js
import * as THREE from "three";
window.THREE = THREE;

import ThreeSDK from "@/ThreeSDK/src/index";
window.MT = null;

onMounted(() => {
  window.MT = new ThreeSDK({ id: "threeBox" });

  MT.scene.background = new THREE.CubeTextureLoader().load([
    "Three/data/SkyBox/px.png",
    "Three/data/SkyBox/nx.png",
    "Three/data/SkyBox/py.png",
    "Three/data/SkyBox/ny.png",
    "Three/data/SkyBox/pz.png",
    "Three/data/SkyBox/nz.png",
  ]);

  // 初始化辅助器
  MT.Helper.addAxesHelper();
});
