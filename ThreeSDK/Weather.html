addRain 添加雨滴效果到场景中 ​

  • @param opt
参数名type描述默认值
rainCountNumber雨滴的数量，20000
rangeNumber粒子特效范围,范围越大 rainCount 的值应该越大60
velocityNumber雨滴的初始速度0.03
  • @returns
返回值type描述
rainObject返回创建的雨滴效果实例,实例中包含更新特效属性的 updateAttr 方法，需要传入属性名和值
js
let weatherEffect = null;

/**雨天 */
const rainEffect = () => {
  if (weatherEffect?.name) {
    destoryEffect();
  }

  weatherEffect = MT.Weather.addRain({ velocity: 0.04 });
};

/**添加粒子数量 */
const updateMeshCount = () => {
  if (weatherEffect) {
    weatherEffect.updateAttr("count", 50000);
  }
};

/**增加动画速度 */
const updateAnimationSpeed = () => {
  if (weatherEffect) {
    weatherEffect.updateAttr("velocity", 1);
  }
};

addCloud 添加云层效果到场景中，实现阴天、多云效果需配合天空盒使用 ​

  • @param
参数名type描述默认值
countNumber云的数量260
rangeNumber云层特效范围,范围越大 count 的值应该越大30000
speedNumber云的初始速度5
heightNumber云的高度1000
cloudColorNumber云的颜色0x808080
typeNumber云层类型，0 为阴天，1 为多云0
  • @returns
返回值type描述
cloudsObject返回创建的云层效果实例
js
let weatherEffect = null;

// ...替换对应天空盒

const snowEffect = () => {
  if (weatherEffect?.name) {
    destoryEffect();
  }

  /**阴天 */
  weatherEffect = MT.Weather.addCloudy({ type: 0, speed: 2, cloudColor: 0x808080 });

  /**多云 */
  // weatherEffect = MT.Weather.addCloudy();
};

destoryEffect 销毁特效 ​

  • @param

if (weatherEffect?.name) {
  MT.Weather.destoryWeather(weatherEffect.name);
  weatherEffect = null;
