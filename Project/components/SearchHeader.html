搜索重置按钮点击 ​
javascript
// 普通搜索搜索
const handleSearch = val => {
    pageInfor.value.pageNum = 1;
    searchParams.value = val;
    _getParksList()
}

// 带时间搜索 【因为时间的keyword 和实际查询的key都可以自定义 就不在组件内处理 在查询时处理 time是配置中的keyword】
const handleSearch = val => {
  let cloneObj = val ? JSON.parse(JSON.stringify(val)) : val;
  if (cloneObj) {
    cloneObj["beginTime"] = "";
    cloneObj["endTime"] = "";
  }
  if (val && Array.isArray(val.time) && val.time.length === 2) {
    cloneObj.beginTime = val.time[0];
    cloneObj.endTime = val.time[1];
    delete cloneObj.time;
  }

  searchParams.value = cloneObj;
  pageInfor.value.pageNum = 1;
  _agpSubjectuserList();
};

// 重置
const handleReset = () => {
	pageInfor.value.pageNum = 1;
    searchParams.value = null;
    _getParksList();
}
